Nim语言笔记

Oct 8, 2019 14:05 · 495 words · 1 minute read Nim语言 

#创建常量(初始化后部不可变)
let a=10;

#创建常量(编译期计算定值)
const a=1;

#创建变量
var a=10

#创建变量设置类型 int float string
let a:int=10

#用户输入
var name=readLine(stdin)

#数组
var x=[1,2,3,4,5]
x[0]=2
echo x[0..2]



#空类型是nil
nil

#转为字符串用$号
$x

#忽略，类似pass
discard

#定义新数据类型
type
    xx

#枚举类型(只允许特定值)
type
    E = enum
        x,y,z
let a=x

#范围类型
type
    x=range[0..5]

#集合类型
let x={'a','b','c'}
let y:set[char]={'a','b','c'}


#数组  
let x=[1,2,3,4,5]
let y:array[0..5,int]=[1,2,3,4,5]
low(x),high(x)


#序列
let x = @[1,2,3,4,5]
let y:seq[int] = @[1,2,3,4,5,6]

# 只能作为函数参数
#开放数组
proc hi(x:openArray[string])=
    for i in x:
        echo i

let x=["a","b","c"]
hi(x)

#开放序列
var x:seq[string]


#切片操作
x[1..10]
x[10..^1]

#对象
type
    xx=object
        x:类型
var y=xx(x:xxx)

#元组
# type
#     Dict=tuple
#         a:int
#         b:int
let x=(a:1,b:2)

#跟踪指针
var x:ref int
x.new
x[]=10
echo x.repr
# type
#     Node = ref object
#       le, ri: Node
#       data: int
#   var
#     n: Node
#   new(n)
#   n.data = 9 

#程序类型
proc xx(x:int)=xxx

proc yy(zz:proc (x:int))=
    xxx
    zz(xx)

#判断
if xx：

elif xx:

else:
    xx


##case语句
case xx：

of xx：

of xx：

else：
    xx

#while循环
while xx:
    xx

#for循环
for i in 1..100:
    echo i

for i in 1..<100:
    echo i

for i,j in xx:
    echo xx

#块语句
block xx:
    xx
#跳出块
break xx


#和if差不多，但在编译期间计算
when xx:

elif xx:

else:
    xx

#函数
proc xx(变量:类型):返回类型 =
    xxx
    result = xx
    (return xx)

#传入可变函数参数
proc xx(变量:var 类型):返回类型 =
    xxx

#泛型
proc foo[T](x:T,y:T):T=
    return x+y

echo foo(10,2)

#高级类型示例
let a=(1,2,3)
let b=[1,2,3]

let y= {1,2,3}

echo b.typeof

echo y.typeof
echo z.typeof
